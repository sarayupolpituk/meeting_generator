25' ====================== MAIN MACRO =======================
26Sub CreateMeetingFromTemplate()
27
28 ' === STEP 1: TITLE ===
29 subject = InputBox("Meeting Title:", "Step 1 - Title", "Project Discussion")
30 ' Subject line = Title — Team[suffix] (auto-appended)
31 ' === STEP 2: RECIPIENT TEAM ===
32 recipientTeam = InputBox("Recipient Team:", "Step 2", "xxxx")
33 ' === STEP 3: SHORT DESCRIPTION ===
34 description = InputBox("Short description:", "Step 3", "Monthly Progress Review")
35 ' === STEP 4: DATE ===
36 ' "7 April 2026" / "07/04/2026" / "2026-04-07" / "April 7, 2026"
37 ' Wrong format → re-prompt (no crash). Default = tomorrow.
38 ' === STEP 5: START & END TIME ===
39 ' No AM/PM → 24hr out.  With AM/PM → 12hr out.  0900 accepted.
40 ' Start default: "09:00" End default: "10:00"
41 ' === STEP 6: LOCATION ===
42 location = InputBox("Location:", "Step 6", "MS Teams")
43 ' "teams" in text → auto-tries Teams join link via COM add-in
44 ' === STEP 7: AGENDAS (1–5) ===
45 ' Leave blank to skip. Numbers auto-adjust — no gaps.
46 ag(5) = InputBox("Agenda 5:", "Step 7 [5/5]", "Action items for proceed")
47 ' === WRITE BODY via WordEditor (bold Date & Time labels) ===
48 oSel.TypeText "Dear " & recipientTeam & " Team,"
49 oSel.TypeText "I would like to appoint for meeting of " & description
50 ' Date/Time: oSel.Font.Bold = True → label bold, False → value normal
51 ' === SIGN-OFF ===
52 oSel.TypeText "Please join in mentioned time."
53 oSel.TypeText "Best Regards,"
54 oSel.TypeText "Sarayu P."
55 ' Prefix: "Eng. Sarayu P." / "Dr. Sarayu P." Suffix: "Sarayu P., PE"
56
57End Sub
