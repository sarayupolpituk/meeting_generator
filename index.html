28' ====================== MAIN MACRO ==========================
29Sub CreateMeetingFromTemplate()
30
31 ' === STEP 1: TITLE ===
32 subject = InputBox("Meeting Title:", "Step 1 - Title",
33 "Project Discussion")
34 ' === STEP 2: RECIPIENT TEAM ===
35 recipientTeam = InputBox("Recipient Team:", "Step 2 - Recipient",
36 "xxxx")
37 ' === STEP 3: SHORT DESCRIPTION ===
38 description = InputBox("Short description:", "Step 3 - Description",
39 "Monthly Progress Review")
40 ' === STEP 4: DATE ===
41 ' ✅ "7 April 2026" / "07/04/2026" / "2026-04-07" / "April 7, 2026"
42 ' Wrong format → re-prompt (no crash). Default = tomorrow.
43 ' === STEP 5: START & END TIME ===
44 ' No AM/PM typed → output: 15:00 (24-hr)
45 ' With AM/PM typed → output: 03:00 PM (12-hr)
46 ' 0900 accepted (no colon). Start default: "09:00" End: "10:00"
47 ' === STEP 6: LOCATION ===
48 location = InputBox("Location:", "Step 6 - Location",
49 "MS Teams")
50 ' Contains "teams" → auto-tries Teams join link via COM add-in
51 ' === STEP 7: AGENDAS (1–5) ===
52 ' Leave blank to skip. Numbers auto-adjust — no gaps.
53 ag(5) = InputBox("Agenda 5:", "Step 7 [5/5]",
54 "Action items for proceed")
55 ' === WRITE BODY via WordEditor (bold supported) ===
56 oSel.TypeText "Dear " & recipientTeam & " Team,"
57 oSel.TypeText "I would like to appoint for meeting of " & description
58 ' Date / Time labels: oSel.Font.Bold = True/False
59 ' === SIGN-OFF ===
60 oSel.TypeText "Please join in mentioned time."
61 oSel.TypeText "Best Regards,"
62 oSel.TypeText "Sarayu P."
63 ' Prefix: "Eng. Sarayu P." / "Dr. Sarayu P."
64 ' Suffix: "Sarayu P., PE" / "Sarayu P., PMP"
65
66End Sub
